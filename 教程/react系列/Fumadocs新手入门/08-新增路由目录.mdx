### 2. 配置内容来源 source.config.ts

```ts
import {
  defineConfig,
  defineDocs,
  frontmatterSchema,
  metaSchema,
} from "fumadocs-mdx/config";

const createDefineDocs = (dir: string) => {
  return defineDocs({
    dir: dir,
    docs: {
      schema: frontmatterSchema,
    },
    meta: {
      schema: metaSchema,
    },
  });
};
export const docs = createDefineDocs("./内容/文档");
export const tutorials = createDefineDocs("./内容/教程");
export const nova = createDefineDocs("./内容/教程/laravel/nova");

export default defineConfig({
  mdxOptions: {},
});
```
