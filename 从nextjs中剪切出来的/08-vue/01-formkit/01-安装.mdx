```ts filename="❻formkit配置" switcher
// 先新建配置文件 formkit.config.ts
// pnpm dlx formkit@latest theme
// pnpm add @formkit/icons
// pnpm add @formkit/i18n
import { genesisIcons } from '@formkit/icons'
import { rootClasses } from './formkit.theme'
import { zh, en } from '@formkit/i18n'
import { defaultConfig } from '@formkit/vue'

export default defaultConfig({
  locales: { zh, en },
  locale: 'zh',
  icons: {
    ...genesisIcons,
  },
  config: {
    rootClasses,
  },
})
```

```ts filename="❼修改main.ts" switcher
import './assets/main.css'
import { createApp } from 'vue'
import { createPinia } from 'pinia'
import App from './App.vue'
import router from './router'
import { plugin, defaultConfig } from '@formkit/vue' //[!code ++] 导入
import config from '../formkit.config.ts' //[!code ++] 导入formkit配置文件
const app = createApp(App)
app.use(plugin, defaultConfig(config)) //[!code ++] 插件
app.use(createPinia())
app.use(router)
app.mount('#app')
```
